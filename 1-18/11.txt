n = int(input())
a, b = [0] * 80, [0] * 80
ans = 0
for i in range(n):
    sch = int(input())
    if sch > 50:
        b[sch % 80] += 1
    else:
        a[sch % 80] += 1

ans += b[0] * (b[0] - 1) // 2 + a[0] * b[0]
ans += b[40] * (b[40] - 1) // 2 + a[40] * b[40]
for i in range(1, 40):
    ans += b[i] * b[80 - i] + b[i] * a[80 - i] + a[i] * b[80 - i]

print(ans)
